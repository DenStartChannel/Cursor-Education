Basic

git config --global user.name blabla - ім'я користувача
git config --global user.email blabal@bla - електрона пошта користувача
git init - дозволяє створити новий репозиторій.
git status - показує статус репозиторію
git add - додає зміни у файл в репозиторій
git commit - комітить всі зміни і зберігає їх в історію версій
git log - показує історію комітів
git checkout - дозволяє перейти до гілки яка уже створена або створити і прейти в неї.
git branch - дозволяє додавати і видаляти нову гілку.
git push - надсилає зміни з локального репозиторію у відалений репозиторій
git pull - отримує остані зміни з відаленого на локальний репозиторій
git merge - об'єднює гілки з батьківською гілкою
git clone - створює локальну робочу копію твого коду з репозиторія.

.gitignore

.gitignore файл - в ньому записують назви файлів які не попадуть в коміти

advanced

git revert - створює коміт який скасовує зміни в останньому коміту
git reset - скидає останні коміти
git rebase - з'єднює без історії
git cherry-pick - перенесення одного коміта в іншу
git stash - перемістити зміни в іншу гілку
